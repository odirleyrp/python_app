#### Procedimento ISTIO #########

# Versão do Istio correspondente ao do EKS
https://istio.io/latest/docs/releases/supported-releases/

# Fazer download da versão 1.25 do ISTIO
curl -L https://istio.io/downloadIstio | sh -

# copiar o binario para o path dos binarios linux
sudo cp istio-1.25.2/bin/istioctl /usr/local/bin

# Validar Versão do ISTIO
istioctl version

# Instalar o Istio no cluster
istioctl install --set profile=demo -y

# Validar se os Pods do Istio foram instalados 
k get pods -n istio-system

# Injetar label no namespace
kubectl label namespace <nome_namespace> istio-injection=enabled

# Instalar o GW do Istio 
k apply -f gw.yaml

# Aplicar o VirtualService
k apply -f vs.yaml


####
#para visiualizar a dinamica do Istio podemos ver pelo Kiali 
#instale primeiro o Prometheus
kubectl apply -f https://raw.githubusercontent.com/istio/istio/release-1.25/samples/addons/prometheus.yaml

# Em seguida instale o Kiali
kubectl apply -f https://raw.githubusercontent.com/istio/istio/release-1.25/samples/addons/kiali.yaml

# Visualizar o dash do Kiali
istioctl dashboard kiali
